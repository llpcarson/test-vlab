617  end if
618  end if

691  wgt1 = 0.5

699  mye = emelt * rhavg ** efac

701  else if ((twq(k1) <= twmelt).and.(twq(k1) < twmelt)) then ! mix
702  rhavg = (rhq(k1)+rhtop) * 0.5
703  dtavg = twmelt - (twq(k1)+twtop) * 0.5
704  dpk = log(pq(k1)/ptop) !lin dpk=pq(k1)-ptop
705 ! mye = emelt*(1.0-(1.0-rhavg)*efac)
706  mye = emelt * rhavg ** efac
707  icefrac = icefrac + dpk * dtavg / mye
708  else ! mix where tw curve crosses twmelt in layer
709  if (twq(k1) == twtop) go to 40 ! both equal twmelt, nothing h
710  wgt1 = (twmelt-twq(k1)) / (twtop-twq(k1))
711  wgt2 = 1.0 - wgt1
712  rhavg = rhtop + wgt2 * (rhq(k1)-rhtop) * 0.5
713  dtavg = (twmelt-twtop) * 0.5
714  dpk = wgt2 * log(pq(k1)/ptop) !lin dpk=wgt2*(pq(k1)-ptop)

716  mye = emelt * rhavg ** efac
717  icefrac = icefrac + dpk * dtavg / mye

719  if (icefrac <= 0.0) then
720 ! goto 1020

722  wgt1 = (twice-twq(k1)) / (twtop-twq(k1))
723  dtavg = twmelt - (twq(k1)+twice) * 0.5
724  else
725  dtavg = (twmelt-twq(k1)) * 0.5
726  end if
727  rhavg = rhq(k1) + wgt1 * (rhtop-rhq(k1)) * 0.5
728  dpk = wgt1 * log(pq(k1)/ptop) !lin dpk=wgt1*(pq(k1)-ptop)
729 ! mye = emelt*(1.0-(1.0-rhavg)*efac)
730  mye = emelt * rhavg ** efac

732  end if

735

907  subroutine calwxt_bourg(lm,lp1,rn,g,t,q,pmid,pint,zint,ptype)
908  implicit none
909 !
910 ! input:
911  integer,intent(in) :: lm,lp1
912  real,intent(in) :: g,rn(2)
913  real,intent(in), dimension(lm) :: t, q, pmid
914  real,intent(in), dimension(lp1) :: pint, zint
915 !
916 ! output:
917  integer, intent(out) :: ptype
918 !
919  integer ifrzl,iwrml,l,lhiwrm
920  real pintk1,areane,tlmhk,areape,pintk2,surfw,area1,dzkl,psfck,r1,r2
921 !
922 ! initialize weather type array to zero (ie, off).
923 ! we do this since we want ptype to represent the
924 ! instantaneous weather type on return.
925 !
926  ptype = 0
927  psfck = pint(lm+1)
928 
929 ! find the depth of the warm layer based at the surface
930 ! this will be the cut off point between computing
931 ! the surface based warm air and the warm air aloft
932 !
933 ! lowest layer t
934 !
935  tlmhk = t(lm)
936  iwrml = lm + 1
937  if (tlmhk >= 273.15) then
938  do l = lm, 2, -1
939  if (t(l) >= 273.15 .and. t(l-1) < 273.15 .and. &
940  & iwrml == lm+1) iwrml = l
941  end do
942  end if
943 !
944 ! now find the highest above freezing level
945 !
946  lhiwrm = lm + 1
947  do l = lm, 1, -1
948 ! gsm added 250 mb check to prevent stratospheric warming situations
949 ! from counting as warm layers aloft
950  if (t(l) >= 273.15 .and. pmid(l) > 25000.) lhiwrm = l
951  end do
952 
953 ! energy variables
954 ! surfw is the positive energy between the ground
955 ! and the first sub-freezing layer above ground
956 ! areane is the negative energy between the ground
957 ! and the highest layer above ground
958 ! that is above freezing
959 ! areape is the positive energy "aloft"
960 ! which is the warm energy not based at the ground
961 ! (the total warm energy = surfw + areape)
962 !
963 ! pintk1 is the pressure at the bottom of the layer
964 ! pintk2 is the pressure at the top of the layer
965 ! dzkl is the thickness of the layer
966 ! ifrzl is a flag that tells us if we have hit
967 ! a below freezing layer
968 !
969  pintk1 = psfck
970  ifrzl = 0
971  areane = 0.0
972  areape = 0.0
973  surfw = 0.0
974 
975  do l = lm, 1, -1
976  if (ifrzl == 0 .and. t(l) <= 273.15) ifrzl = 1
977  pintk2 = pint(l)
978  dzkl = zint(l)-zint(l+1)
979  if (t(l) >= 273.15 .and. pmid(l) > 25000.) then
980  area1 = log(t(l)/273.15) * g * dzkl
981  if (l < iwrml) then
982  areape = areape + area1
983  else
984  surfw = surfw + area1
985  endif
986  elseif (l > lhiwrm) then
987  area1 = log(t(l)/273.15) * g * dzkl
988  areane = areane + abs(area1)
989  endif
990  pintk1 = pintk2
991  enddo
992 
993 !
994 ! decision tree time
995 !
996  if (areape < 2.0) then ! very little or no positive energy aloft, check for
997  ! positive energy just above the surface to determine rain vs. snow
998  if (surfw < 5.6) then ! not enough positive energy just above the surface snow = 1
999  ptype = 1
1000  else if (surfw > 13.2) then ! enough positive energy just above the surface rain = 8
1001  ptype = 8
1002  else ! transition zone, assume equally likely rain/snow
1003  ! picking a random number, if <=0.5 snow
1004  r1 = rn(1)
1005  if (r1 <= 0.5) then ! snow = 1
1006  ptype = 1
1007  else ! rain = 8
1008  ptype = 8
1009  end if
1010  end if
1011 !
1012  else ! some positive energy aloft, check for enough negative energy
1013  ! to freeze and make ice pellets to determine ip vs. zr
1014 
1015  if (areane > 66.0+0.66*areape) then
1016 ! enough negative area to make ip,
1017 ! now need to check if there is enough positive energy
1018 ! just above the surface to melt ip to make rain
1019  if (surfw < 5.6) then ! not enough energy at the surface to melt ip ice pellets = 2
1020  ptype = 2
1021  elseif (surfw > 13.2) then ! enough energy at the surface to melt ip rain = 8
1022  ptype = 8
1023  else ! transition zone, assume equally likely ip/rain picking a random number, if <=0.5 ip
1024  r1 = rn(1)
1025  if (r1 <= 0.5) then ! ice pellets = 2
1026  ptype = 2
1027  else ! rain = 8
1028  ptype = 8
1029  end if
1030  end if
1031  elseif (areane < 46.0+0.66*areape) then
1032 ! not enough negative energy to refreeze, check surface temp to determine rain vs. zr
1033  if (tlmhk < 273.15) then ! freezing rain = 4
1034  ptype = 4
1035  else ! rain = 8
1036  ptype = 8
1037  end if
1038  else
1039 ! transition zone, assume equally likely ip/zr picking a random number, if <=0.5 ip
1040  r1 = rn(1)
1041  if (r1 <= 0.5) then
1042 ! still need to check positive energy just above the surface to melt ip vs. rain
1043  if (surfw < 5.6) then ! ice pellets = 2
1044  ptype = 2
1045  else if (surfw > 13.2) then ! rain = 8
1046  ptype = 8
1047  else
1048 ! transition zone, assume equally likely ip/rain picking a random number, if <=0.5 ip
1049  r2 = rn(2)
1050  if (r2 <= 0.5) then ! ice pellets = 2
1051  ptype = 2
1052  else ! rain = 8
1053  ptype = 8
1054  end if
1055  end if
1056  else
1057 ! not enough negative energy to refreeze, check surface temp to determine rain vs. zr
1058  if (tlmhk < 273.15) then ! freezing rain = 4
1059  ptype = 4
1060  else ! rain = 8
1061  ptype = 8
1062  end if
1063  end if
1064  end if
1065  end if
1066 !
1067  return
1068  end
